DROP DATABASE attackDB;
CREATE DATABASE IF NOT EXISTS attackDB;
USE attackDB;

CREATE TABLE IF NOT EXISTS urls (
    id INT AUTO_INCREMENT,
    url LONGTEXT,
    PRIMARY KEY (id)
);


CREATE TABLE IF NOT EXISTS payloads_Reflectedxss (
    id INT AUTO_INCREMENT,
    payload LONGTEXT,
    PRIMARY KEY (id)
);
CREATE TABLE IF NOT EXISTS payloads_Storedxss (
    id INT AUTO_INCREMENT,
    payload LONGTEXT,
    PRIMARY KEY (id)
);
CREATE TABLE IF NOT EXISTS payloads_Domxss (
    id INT AUTO_INCREMENT,
    payload LONGTEXT,
    PRIMARY KEY (id)
);

CREATE TABLE IF NOT EXISTS payloads_sqli (
    id INT AUTO_INCREMENT,
    payload LONGTEXT,
    PRIMARY KEY (id)
);

CREATE TABLE IF NOT EXISTS payloads_or (
    id INT AUTO_INCREMENT,
    payload LONGTEXT,
    PRIMARY KEY (id)
);
